                """
                Return the digit at a particular place in a 
                base 10 number

                Parameters
                ----------
                x: int
                    An integer
                place: int
                    Which place to look at, where 0 is the 1s place,
                    1 is the 10s place, 2 is the 100s place, etc
                """
                digit = 0
                s = "{}".format(x)
                if place < len(s):
                    digit = int(s[-(1+place)])
                return digit
